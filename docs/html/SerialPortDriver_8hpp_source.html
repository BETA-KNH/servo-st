123 std::vector<uint8_t> read(std::size_t expectedBytes) const
124 {

133 struct pollfd pfd{fd_, POLLIN, 0};
134 int r = ::poll(&pfd, 1, timeoutMs_);
135 if (r < 0) {
136 if (errno == EINTR) continue; // signal — retry

138 }
139 if (r == 0) break; // timeout expired
140 if (!(pfd.revents & POLLIN)) break; // unexpected event
141
142 ssize_t n = ::read(fd_,

145 if (n < 0) {

147 break;
148 }
149 if (n == 0) break;
150 received += static_cast<std::size_t>(n);

193 void open(const std::string& port, int baud, int timeoutMs)
194 {
195 // Validate baud rate before touching the fd — throws for unknown values
196 const speed_t speed = baudToConstant(baud);
197
198 // O_NOCTTY — don't make this the controlling terminal
199 // O_NDELAY — don't block on open even if DCD is not asserted
200 fd_ = ::open(port.c_str(), O_RDWR | O_NOCTTY | O_NDELAY);
201 if (fd_ < 0) return; // caller checks isOpen()
202
203 // Restore blocking behaviour for reads
204 ::fcntl(fd_, F_SETFL, 0);
205
206 // ---- termios configuration ----
207 struct termios tty{};
208 if (::tcgetattr(fd_, &tty) != 0) {
209 ::close(fd_);
210 fd_ = -1;
211 return;
212 }
213
214 // Raw mode: no line processing, no signals, no special characters
215 ::cfmakeraw(&tty);
216
217 // 8 data bits, no parity, 1 stop bit (8N1)
218 tty.c_cflag &= ~static_cast<tcflag_t>(CSIZE);
219 tty.c_cflag |= CS8;
220 tty.c_cflag &= ~static_cast<tcflag_t>(PARENB);
221 tty.c_cflag &= ~static_cast<tcflag_t>(CSTOPB);
222
223 // Disable hardware flow control
224 tty.c_cflag &= ~static_cast<tcflag_t>(CRTSCTS);
225
226 // Enable receiver, local mode (ignore modem control lines)
227 tty.c_cflag |= (CLOCAL | CREAD);
228
229 // Baud rate (already resolved above)
230 ::cfsetispeed(&tty, speed);
231 ::cfsetospeed(&tty, speed);
232
233 // Non-blocking reads — poll() in read() provides the timeout.
234 // VMIN=0, VTIME=0: read() returns immediately with whatever is buffered.
235 tty.c_cc[VMIN] = 0;
236 tty.c_cc[VTIME] = 0;
237
238 if (::tcsetattr(fd_, TCSANOW, &tty) != 0) {
239 ::close(fd_);
240 fd_ = -1;
241 }
242 }
