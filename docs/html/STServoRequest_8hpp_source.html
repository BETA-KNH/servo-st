148 static std::vector<uint8_t> buildPacket(uint8_t id,
149 STServo::Instruction instr,
150 const std::vector<uint8_t>& params)
151 {
152 const uint8_t len = static_cast<uint8_t>(2 + params.size());
153
154 std::vector<uint8_t> packet;
155 packet.reserve(5 + params.size()); // 2×header + id + len + instr + params + cksum
156 packet.push_back(STServo::HEADER);
157 packet.push_back(STServo::HEADER);
158 packet.push_back(id);
159 packet.push_back(len);
160 packet.push_back(static_cast<uint8_t>(instr));
161 packet.insert(packet.end(), params.begin(), params.end());
162
163 // Checksum covers every byte from ID onward (excludes the two 0xFF headers)
164 uint8_t sum = 0;
165 for (std::size_t i = 2; i < packet.size(); ++i)
166 sum = static_cast<uint8_t>(sum + packet[i]);
167 packet.push_back(static_cast<uint8_t>(~sum & 0xFFu));
168
169 return packet;
170 }
